docker-compose up -d            # Запускает контейнеры в фоновом режиме

docker-compose stop             # Останавливает контейнеры

docker-compose down             # Останавливает и удаляет контейнеры, сети и образы (тома не удаляются) - т.е. база там будет жить

docker-compose down --volumes   # Останавливает и удаляет контейнеры, сети, образы и тома - а такое убьет и волиум с базой


только такое не делай - docker-compose down --volumes -это убивает волиумы
